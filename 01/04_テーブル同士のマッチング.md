# テーブル同士のマッチング
CASEは式を評価できるため、LIKEなどの述語群を使用できる  
中でもINとEXISTSは引数にサブクエリを指定できるため、表現力に幅ができる  
サブクエリで主キーのインデックスを利用できたりするため、EXISTSの方がパフォーマンスはよい
``` SQL
-- IN
SELECT
	course_name
	,CASE WHEN course_id IN (SELECT course_id FROM OpenCourse WHERE month = '2018-06-01') THEN '〇'
		ELSE  '×'
	 END AS '6月'
	 ,CASE WHEN course_id IN (SELECT course_id FROM OpenCourse WHERE month = '2018-07-01') THEN '〇'
		ELSE  '×'
	 END AS '7月'
	 ,CASE WHEN course_id IN (SELECT course_id FROM OpenCourse WHERE month = '2018-08-01') THEN '〇'
		ELSE  '×'
	 END AS '8月'
FROM
	CourseMaster

-- EXISTS
SELECT
	course_name
	,CASE WHEN EXISTS (SELECT course_id FROM OpenCourse AS OC WHERE month = '2018-06-01'
		AND OC.course_id = CM.course_id) THEN '〇'
		ELSE  '×'
	 END AS '6月'
	 ,CASE WHEN EXISTS (SELECT course_id FROM OpenCourse AS OC WHERE month = '2018-07-01'
		AND OC.course_id = CM.course_id) THEN '〇'
		ELSE  '×'
	 END AS '7月'
	 ,CASE WHEN EXISTS (SELECT course_id FROM OpenCourse AS OC WHERE month = '2018-08-01'
		AND OC.course_id = CM.course_id) THEN '〇'
		ELSE  '×'
	 END AS '8月'
FROM
	CourseMaster AS CM
```
